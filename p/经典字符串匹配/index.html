int BFMatch(char s[], int len_s, char p[], int len_p) {
    if (len_s < len_p)
        return -1;
    int i = 0, j = 0;
    // ‰πãÂåπÈÖçsÁöÑÂâçlen_s-len_p+1‰∏™
    // Â¶ÇÊûús[len(s)-len(p)] != p[0]
