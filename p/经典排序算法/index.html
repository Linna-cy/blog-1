#include <stdio.h>

void insertionSort(int arr[], int len)
{
    for (int i = 0 ; i < len - 1 ; i++)
    {
        int current = arr[i + 1];
        int j = i;
        while (j >= 0 && arr[j] > current)
        {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = current;
    }
}
